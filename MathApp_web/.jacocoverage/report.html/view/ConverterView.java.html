package view;

/**
 * View create response to servlet request for html page
 * @author Paulina
 * @version 1.0
 */
public class ConverterView {
    
    /**
     * Contains text field 
     */
    private String number;
    
    /**
     * Contains result
     */
    private String result;
    
    /**
     * Contains number of conversions
     */
    private int numberOfConversions;
    
    /**
     * Contains response from servlet
     */
    private String response;
    
    /**
     * Conatins optional message
     */
    private String message;

    /**
     * Contains history of conversion
     */
    private String history;
    
    /**
     * Argumentless constructor
     */
    
    
    public ConverterView()
    {
        number = "";
        result = "";
        numberOfConversions = 0;
    }
    
    /**
     * Creates response from serverlet
     */
    public void createPage()
    {
        response = "<!DOCTYPE html>\n"
                + "<html>\n"
                + "     <head>\n"
                + "         <title>HEX TO DEC CONVERTER</title>\n"
                + "         <meta charset=\"UTF-8\">\n"
                + "         <meta name=\"viewport\" content=\"width=device-width, initial-scale=1.0\">\n"
                + "     </head>\n"
                + "     <body>\n"
                + "     <div id=\"view\">\n"
                + "         <h1 id=\"label1\">Please enter number to convert below, and choose type of conversion! </h1>\n"
                + "         <form action=\"Convert\" method=\"POST\">\n"
                + "             <input name=\"number\" type=\"text\" value=\"\" name=\"number\" >\n"
                + "             <button name=\"h2d\" type=\"submit\">Hex to Dec</button>\n"
                + "             <button name=\"d2h\" type=\"submit\">Dec to Hex</button>\n"
                + "         </form>"
                + "         <h2 id=\"label2\">Result: " + result + "</h2>\n"
                + "         <form action=\"History\" method=\"POST\">\n"
                + "         <input name=\"history\" type=\"submit\" value=\"History\" />\n"
                + "         </form>\n"
                + "     <h3 id=\"historyList\">" + history + "</h3>\n"
                + "     <div id=\"message\"> " + message + "</div>\n"
                + "     <div id=\"cookie\"> You converted " + numberOfConversions + " numbers on this page!</div>\n" 
                + "     </div>\n"
                + "     </body>\n"
                + "</html>";
    }
    
    /**
     * Returns response as html page
     * @return response
     */
    public String getPage()
    {
        return response;
    }
    
    public void setNumberOfConversions(int val)
    {
        this.numberOfConversions = val;
    }
    
    public void setResult(String val)
    {
        this.result = val;
    }
    
    public void setMessage(String mes)
    {
        this.message = mes;
    }
    
    public void setHistory(String his)
    {
        this.history = his;
    }
    
}
